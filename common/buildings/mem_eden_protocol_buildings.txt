building_mem_eden_protocol_seeder_1 = {
	base_buildtime = @b2_time
	base_cap_amount = 1
	category = unity
	can_be_disabled = no
	upgrades = {
		"building_mem_eden_protocol_seeder_2"
	}
	potential = {
		is_planet_class = pc_habitat
		exists = owner
		owner = {
			is_country_type = default
			has_origin = origin_mem_eden_protocol
		}
		solar_system = {
			any_system_planet = {
				habitable_planet_not_urban = yes
				OR = {
					NOT = {
						exists = owner
					}
					AND = {
						exists = owner
						owner = {
							is_country_type = primitive
						}
					}
				}
			}
		}
	}
	destroy_trigger = {
		OR = {
			NOT = {
				exists = owner
			}
			owner = {
				NOT = {
					has_origin = origin_mem_eden_protocol
				}
			}
			solar_system = {
				NOT = {
					any_system_planet = {
						habitable_planet_not_urban = yes
						OR = {
							NOT = {
								exists = owner
							}
							AND = {
								exists = owner
								owner = {
									is_country_type = primitive
								}
							}
						}
					}
				}
			}
		}
	}
	on_destroy = {
		solar_system = {
			every_system_planet = {
				limit = {
					exists = owner
					owner = {
						is_country_type = primitive
						OR = {
							has_country_flag = machine_age
							has_country_flag = atomic_age
							has_country_flag = early_space_age
						}
					}
				}
				owner = {
					random_situation = {
						limit = {
							is_situation_type = pre_ftl_tech_progress_situation
						}
						remove_situation_flag = tech_frozen
					}
				}
			}
		}
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
		}
		upkeep = {
			energy = 5
		}
	}
	planet_modifier = {
		job_mem_seeder_drone_add = 2
	}
	triggered_planet_modifier = {
		job_mem_caretaker_drone_add = 1
		mult = value:mem_eden_protocol_caretaker_value
	}
	triggered_planet_modifier = {
		potential = {
			owner = {
				has_valid_civic = civic_machine_servitor
			}
			
		}
		modifier = {
			job_bio_trophy_add = 5
		}
	}
	ai_weight = {
		weight = 1000
	}
	#Always build the pop to work it can come later.
}
